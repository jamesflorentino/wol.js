        generate: function(columns, rows) {
            var row, col, tile;
            this.columns = columns;
            this.rows = rows;
            for(col = 0; col < columns; col++) {
                this._keys[col] = {};
                for(row = 0; row < rows; row++) {
                    tile = new Tile(col, row);
                    this._keys[col][row] = tile;
                    this._tiles.push(tile);
                }
            }
        },
